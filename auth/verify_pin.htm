async function submitPin(pin) { const res = await fetch("/auth/verify_pin.php", { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded" }, body: "pin=" + encodeURIComponent(pin) }); const data = await res.json(); if (data.ok) { window.location.href = "/app/timeline.php"; } else { alert(data.msg); } }
